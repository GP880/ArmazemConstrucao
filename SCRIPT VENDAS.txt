

SCRIPT VENDAS

CREATE TABLE [dbo].[Venda] (
    [IdVenda]     INT             IDENTITY (1, 1) NOT NULL,
    [NumNotaFisc] INT             NOT NULL,
    [IdItens]     INT             NOT NULL,
    [CodColab]    INT             NOT NULL,
    [ValTot]      DECIMAL (10, 2) NOT NULL,
    [IdColab]     INT             NOT NULL,
    [Desconto]    DECIMAL (10, 2) NULL,
    [Dinheiro]    DECIMAL (10, 2) NULL,
    [CartDeb]     DECIMAL (10, 2) NULL,
    [CartCred]    DECIMAL (10, 2) NULL,
    [Pix]         DECIMAL (10, 2) NULL,
    [Data]        DATE            NULL,
    PRIMARY KEY CLUSTERED ([IdVenda] ASC)
);
